import { OpenAI } from 'openai'; const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY, }); async function generateContent(text) { const prompt = `Analizza questo testo educativo e genera un JSON con la seguente struttura ESATTAMENTE: { "summary_short": "Un riassunto di 2-3 righe", "summary_long": "Un riassunto dettagliato di 5-7 righe", "highlights": ["punto 1", "punto 2", "punto 3", "punto 4", "punto 5"], "glossary": [ {"term": "termine", "definition": "definizione con contesto"} ], "timeline": [ {"date": "1948", "event": "evento importante"} ], "mind_map": ["concetto 1", "concetto 2", "concetto 3", "concetto 4"], "flashcards": [ {"q": "Domanda 1?", "a": "Risposta 1"} ], "quiz": [ {"q": "Domanda quiz?", "options": ["A", "B", "C", "D"], "correct": 0} ] } Testo da analizzare: ${text} IMPORTANTE - LEGGI ATTENTAMENTE: - Rispondi SOLO con JSON valido - highlights: ARRAY DI OGGETTI. Ogni oggetto ha "title" (il punto chiave) e "concepts" (array di 3-5 concetti correlati) Esempio: {"title": "Legge 104/1992", "concepts": ["diritti degli studenti", "inclusione scolastica", "bisogni educativi speciali"]} - glossary: 8-10 termini importanti CON definizioni brevi e contesto - timeline: eventi significativi con date ESATTE (se nel testo) - mind_map: 4 concetti fondamentali e interconnessi - flashcards: 10-12 domande/risposte per apprendimento attivo. Varia il tipo: definizioni, applicazioni, differenze, esempi - quiz: 8-10 domande DIFFICILI e SPECIFICHE (non banali). Ogni domanda deve testare comprensione profonda, non solo memoria. Le opzioni sbagliate devono essere plausibili. Varia il tipo: applicazione, analisi, inferenza - Usa italiano per tutto il contenuto - I quiz devono essere sfidanti ma risolvibili con il testo fornito`; const response = await openai.chat.completions.create({ model: 'gpt-3.5-turbo', messages: [ { role: 'user', content: prompt, }, ], temperature: 0.7, max_tokens: 2000, }); const content = response.choices[0].message.content; // Estrai JSON dalla risposta const jsonMatch = content.match(/\{[\s\S]*\}/); if (!jsonMatch) { throw new Error('Impossibile estrarre JSON dalla risposta'); } return JSON.parse(jsonMatch[0]); } export default async function handler(req, res) { if (req.method !== 'POST') { return res.status(405).json({ error: 'Metodo non consentito' }); } try { const { text } = req.body; if (!text || text.trim().length === 0) { return res.status(400).json({ error: 'Testo PDF vuoto' }); } // Limita il testo a ~5000 caratteri per non superare i token const limitedText = text.substring(0, 5000); const data = await generateContent(limitedText); res.status(200).json(data); } catch (error) { console.error('Errore:', error); res.status(500).json({ error: error.message || 'Errore nella generazione contenuti' }); } }
